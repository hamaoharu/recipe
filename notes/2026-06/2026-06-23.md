 - [id]によるparamsの取得 非同期
 ```
type ParamPageProps = {
    params: Promise<{id: string}>;
},

export default async function ParamPage({ params }: ParamPageProps) {
    //idは分割代入 params.id
    const { id } = await params;
    return <p>No.{id}のページを表示しています。</p>;
}
 ```
